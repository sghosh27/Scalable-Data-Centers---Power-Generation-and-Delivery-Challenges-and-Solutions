"""
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
from pulp import LpMinimize, LpProblem, LpVariable, lpSum, LpBinary, LpContinuous, value

# ----------------------------
# Model Setup and Parameters
# ----------------------------
model = LpProblem("Dynamic_Server_Loading", LpMinimize)

# Sets
I = range(5)   # 5 servers
T = range(24)  # 24 hours

# Toggle Demand Response (Set to True to enable, False to disable)
demand_response = False

# Electricity price ($/kWh) peaks between 12 PM - 2 PM (Hours 12-14)
C = [0.05 + 0.05 * (t % 8) if not (12 <= t <= 14) else 0.80 for t in T]

# Startup and shutdown costs
S = [2 for i in I]
H = [1 for i in I]

# Updated Min and Max Power Limits (STRICTLY ENFORCED)
P_min = [1, 2, 2.5, 1.5, 2]
P_max = [30, 10, 72, 16, 35]  # **Updated Power Limits**

# Updated Efficiency of each server (work done per kW; higher is better)
eta = [0.89, 0.99, 0.75, 0.90, 0.80]

# Upgraded Transmission grid power limit (kW)
P_max_grid = 200

# Minimum up-time (U) and down-time (D) in hours
U = 2
D = 2

# Hourly renewable energy available (kW)
R = [20 + 10 * (t % 8) for t in T]

# Workload demand (kW) that must be met each hour (dynamic)
D_total = [50 + 10 * (t % 8) for t in T]

# ----------------------------
# Demand Response: Identify Peak Hours and Reschedule Loads
# ----------------------------
if demand_response:
    peak_threshold = np.percentile(D_total, 85)  # Define peak threshold (top 15% loads)
    for t in range(len(T) - 2):
        if D_total[t] >= peak_threshold:
            shift_amount = D_total[t] * 0.2  # Shift 20% of load
            D_total[t] -= shift_amount
            D_total[t + 1] += shift_amount * 0.7  # Move 70% to next hour
            D_total[t + 2] += shift_amount * 0.3  # Move 30% to the hour after

# Penalty and balancing parameters:
lambda_balance = 0.1
lambda_penalty = 0.5    # penalize inefficient servers
lambda_reward  = 0.2    # reward loading on efficient servers

# ----------------------------
# Decision Variables
# ----------------------------
x = {(i,t): LpVariable(f"x_{i}_{t}", cat=LpBinary) for i in I for t in T}  # Server on/off
y = {(i,t): LpVariable(f"y_{i}_{t}", cat=LpBinary) for i in I for t in T}  # Startup indicator
z = {(i,t): LpVariable(f"z_{i}_{t}", cat=LpBinary) for i in I for t in T}  # Shutdown indicator
P_var = {(i,t): LpVariable(f"P_{i}_{t}", lowBound=0, upBound=P_max[i], cat=LpContinuous) for i in I for t in T}  # Power consumption (kW)
W = {(i,t): LpVariable(f"W_{i}_{t}", lowBound=0, upBound=P_max[i] * eta[i], cat=LpContinuous) for i in I for t in T}  # Work produced (kW)

# Auxiliary variable for absolute deviation in workload balancing
abs_dev = {(i,t): LpVariable(f"abs_dev_{i}_{t}", lowBound=0, cat=LpContinuous) for i in I for t in T}

# Total workload produced at time t (used in balancing)
W_total = {t: lpSum(W[i,t] for i in I) for t in T}

# ----------------------------
# Objective Function
# ----------------------------
model += lpSum(
    C[t] * P_var[i,t] +
    S[i] * y[i,t] +
    H[i] * z[i,t] +
    lambda_balance * abs_dev[i,t] +
    (lambda_penalty * (1 - eta[i]) - lambda_reward * eta[i]) * P_var[i,t]
    for i in I for t in T
)

# ----------------------------
# Constraints
# ----------------------------
# 1. **Total Power Constraints**
for t in T:
    model += lpSum(P_var[i,t] for i in I) <= R[t] + P_max_grid
    model += lpSum(P_var[i,t] for i in I) <= P_max_grid
    model += lpSum(P_var[i,t] for i in I) >= 0.5  # **Minimum power usage enforced**

# 2. **STRICT Per-Server Power Limits**
for i in I:
    for t in T:
        model += P_var[i,t] <= P_max[i] * x[i,t]  # **STRICT POWER LIMIT**
        model += P_var[i,t] >= P_min[i] * x[i,t]  # **Server must use min power if ON**

# 3. **Workload Efficiency and Strict Upper Bound**
for i in I:
    for t in T:
        model += W[i,t] == eta[i] * P_var[i,t]  # **Work is proportional to power used**
        model += W[i,t] <= P_var[i,t] * eta[i]  # **Ensure work is physically possible**
        model += W[i,t] <= P_max[i] * eta[i] * x[i,t]  # **Prevent excessive work**

# 4. **Startup and Shutdown Constraints**
for i in I:
    for t in T:
        if t > 0:
            model += y[i,t] - z[i,t] == x[i,t] - x[i,t-1]
        model += y[i,t] + z[i,t] <= 1

# 5. **Minimum Uptime and Downtime**
for i in I:
    for t in range(len(T) - U):
        model += lpSum(x[i,k] for k in range(t, t+U)) >= U * y[i,t]
    for t in range(len(T) - D):
        model += lpSum(1 - x[i,k] for k in range(t, t+D)) >= D * z[i,t]

# 6. **Balanced Workload Distribution**
for i in I:
    for t in T:
        model += W[i,t] - (W_total[t] / len(I)) <= abs_dev[i,t]
        model += (W_total[t] / len(I)) - W[i,t] <= abs_dev[i,t]

# 7. **Minimum Workload Demand**
for t in T:
    model += lpSum(W[i,t] for i in I) == D_total[t]  # **Ensures total demand is met**

# ----------------------------
# Solve the Model
# ----------------------------
model.solve()

# ----------------------------
# Debugging & Output
# ----------------------------
for i in I:
    for t in T:
        if P_var[i,t].varValue > P_max[i]:
            print(f" Constraint Violation! Server {i} at hour {t}: {P_var[i,t].varValue:.2f} kW (Limit: {P_max[i]} kW)")

# Generate Heatmap of Server Load
plt.figure(figsize=(12, 6))
ax = sns.heatmap([[P_var[i,t].varValue for t in T] for i in I], annot=True, fmt=".1f", cmap="YlGnBu",
                 xticklabels=[f"Hour {t}" for t in T],
                 yticklabels=[f"Server {i}" for i in I])
ax.set_title("Heatmap of Server Power Load (kW) Over 24 Hours without any Demand Response Participation")
ax.set_xlabel("Time (Hour)")
ax.set_ylabel("Server")
plt.tight_layout()
plt.show()
